# COMANDOS PARA EXECUTAR NO IBM CLOUD SHELL
# Acesse: https://cloud.ibm.com/shell

# 1. Selecionar projeto agroisync
ibmcloud ce project select --name agroisync

# 2. Criar aplicação agroisync-web
ibmcloud ce app create --name agroisync-web --image nginx:alpine --port 80 --cpu 0.25 --memory 0.5Gi

# 3. Verificar status da aplicação
ibmcloud ce app get --name agroisync-web

# 4. Ver logs da aplicação (opcional)
ibmcloud ce app logs --name agroisync-web
